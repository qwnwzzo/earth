    #include <stdio.h>

typedef long long LL;

LL power(LL a, LL b, LL p) {
    LL ans = 1ll % p;
    for (; b; b >>= 1) {
        if (b & 1) ans = ans * a % p;
        a = a * a % p;
    }
    return ans;
}

int main() {
    LL a, b, p; scanf("%lld %lld %lld", &a, &b, &p);
    printf("%lld\n", power(a, b, p));
    return 0;
}
